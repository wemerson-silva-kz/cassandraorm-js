🎯 Scopes & Filters

Reusable query scopes and dynamic filtering.

🏗️ Defining Scopes

// Define scopes in schema
const User = await client.loadSchema('users', {
  fields: { id: 'uuid', name: 'text', status: 'text', age: 'int' },
  key: ['id'],
  scopes: {
    active: () => ({ status: 'active' }),
    adults: () => ({ age: { $gte: 18 } }),
    byDepartment: (dept) => ({ department: dept }),
    recent: (days = 30) => ({
      created_at: { $gte: new Date(Date.now() - days * 24 * 60 * 60 * 1000) }
    })
  }
});

// Using scopes
const activeUsers = await User.scope('active').find();
const recentAdults = await User.scope('adults').scope('recent', 7).find();
