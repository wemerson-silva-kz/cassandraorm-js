⚡ Quick Reference

Basic Usage

import { createClient } from 'cassandraorm-js';

const client = createClient({
  clientOptions: {
    contactPoints: ['127.0.0.1'],
    localDataCenter: 'datacenter1',
    keyspace: 'myapp'
  }
});

await client.connect();

const User = await client.loadSchema('users', {
  fields: {
    id: 'uuid',
    name: 'text',
    email: { type: 'text', unique: true }
  },
  key: ['id']
});

// CRUD operations
const user = await User.save({ name: 'John', email: 'john@example.com' });
const users = await User.find({ name: 'John' });
await User.update({ id: user.id }, { name: 'John Doe' });
await User.delete({ id: user.id });

Advanced Features

// Query Builder
const results = await User.query()
  .where('age', '>=', 18)
  .where('status', '=', 'active')
  .orderBy('created_at', 'desc')
  .limit(10)
  .execute();

// Relationships
const userWithPosts = await User.findOne(
  { id: userId }, 
  { include: ['posts', 'profile'] }
);

// Scopes
User.addScope('active', () => ({ status: 'active' }));
const activeUsers = await User.scope('active').find();

// Hooks
User.addHook('beforeSave', (data) => {
  data.updated_at = new Date();
  return data;
});
