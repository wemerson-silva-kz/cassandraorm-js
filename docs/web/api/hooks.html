  fields: { id: 'uuid', name: 'text', email: 'text', password: 'text' },

    },
    afterSave: (data) => {
      console.log('User saved:', data.id);
      // Send welcome email, etc.
    },
    beforeDelete: (filter) => {
      console.log('Deleting user:', filter);
    }
  }
});
