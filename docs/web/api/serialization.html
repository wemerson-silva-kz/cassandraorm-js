const User = await client.loadSchema('users', {
  fields: { id: 'uuid', name: 'text', email: 'text', password: 'text' },
  key: ['id'],
  serialization: {
    hidden: ['password'], // Hide from JSON output
    casts: {
      created_at: 'date',
      metadata: 'json'
    },
    appends: ['full_name'],
    transforms: {
      full_name: (record) => `${record.first_name} ${record.last_name}`
    }
  }
});

// Serialize for API
const user = await User.findOne({ id: userId });
const serialized = user.toJSON(); // Excludes password, includes full_name
