🔄 Migrations

Schema versioning, migrations, and rollbacks.

🏗️ Migration System

// Create migration
const migration = {
  version: '001',
  up: async (client) => {
    await client.execute(`
      CREATE TABLE users (
        id uuid PRIMARY KEY,
        name text,
        email text,
        created_at timestamp
      )
    `);
    
    await client.execute(`
      CREATE INDEX ON users (email)
    `);
  },
  down: async (client) => {
    await client.execute('DROP INDEX users_email_idx');
    await client.execute('DROP TABLE users');
  }
};

// Run migrations
await client.migrate.up();
await client.migrate.down();
await client.migrate.to('003');
