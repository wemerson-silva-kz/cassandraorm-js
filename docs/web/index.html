üöÄ Quick Start

import { createClient } from 'cassandraorm-js';

const client = createClient({
  clientOptions: {
    contactPoints: ['127.0.0.1'],
    localDataCenter: 'datacenter1',
    keyspace: 'myapp'
  },
  ormOptions: {
    createKeyspace: true,
    migration: 'safe'
  }
});

await client.connect();

// Define schema with all features
const User = await client.loadSchema('users', {
  fields: {
    id: 'uuid',
    email: { type: 'text', unique: true },
    name: 'text',
    created_at: 'timestamp'
  },
  key: ['id'],
  unique: ['email'], // ‚úÖ Automatic validation
  hooks: {
    beforeSave: (data) => {
      data.updated_at = new Date();
      return data;
    }
  },
  scopes: {
    active: () => ({ status: 'active' }),
    recent: (days = 30) => ({
      created_at: { $gte: new Date(Date.now() - days * 24 * 60 * 60 * 1000) }
    })
  }
});

// Use all features automatically
const user = new User({
  id: uuid(),
  email: 'user@example.com',
  name: 'John Doe'
});

await user.save(); // ‚úÖ Automatic unique validation + hooks
